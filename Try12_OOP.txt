# CONTOH STRUKTUR PENCABANGAN IF
#Input data dari pengguna 
nilai_ujian = float(input("Masukkan nilai ujian siswa: "))

#Struktur kontrol percabangan majemuk
if nilai_ujian >= 90:
    predikat = "Sangat Baik (A)"
elif nilai_ujian >=75:
    predikat = "Baik (B)"
elif nilai_ujian >= 60:
    predikat = "Cukup (c)"
else:
    predikat = "Kurang (D) - Memerlukan Remedial"
    
#Menampilkan hasil keputusan 
print("Hasil Evaluasi:", predikat)
